    ```go
    // Package openai implements the Phase-1 ChatModel adapter against
    // github.com/openai/openai-go/v3 (Chat Completions API only per D-01 of
    // .planning/phases/01-walking-skeleton-generate/01-CONTEXT.md).
    //
    // Stream is a Phase-1 stub. Tool calling, embeddings, and structured
    // outputs land in Phases 2/3/4.
    package openai

    import (
        "context"
        "errors"

        "github.com/costa92/llm-agent/llm"
        openai "github.com/openai/openai-go/v3"
    )

    // Compile-time: ChatModel ONLY — Phase 1 ignores tools / embeddings / structured outputs.
    var _ llm.ChatModel = (*OpenAI)(nil)

    // OpenAI is the Chat Completions adapter. Construct via New(opts ...Option).
    type OpenAI struct {
        client *openai.Client
        info   llm.ProviderInfo
    }

    // Generate sends a sync chat completion request and maps the response back
    // into the llm.Response shape. SDK errors are wrapped via wrapErr.
    func (o *OpenAI) Generate(ctx context.Context, req llm.Request) (llm.Response, error) {
        sdkReq := o.toSDKRequest(req)
        completion, err := o.client.Chat.Completions.New(ctx, sdkReq)
        if err != nil {
            return llm.Response{}, wrapErr(err)
        }
        return o.fromSDKResponse(completion), nil
    }

    // Stream is a Phase-1 stub. Streaming lands in Phase 2 (CONF-03; OAI-02).
    func (o *OpenAI) Stream(_ context.Context, _ llm.Request) (llm.StreamReader, error) {
        return nil, errors.New("openai: streaming not implemented in Phase 1; use Generate")
    }

    // Info returns the bound (provider, model, capabilities) tuple. Phase 1
    // adapters report Capabilities.Tools=false even though the SDK supports tools.
    func (o *OpenAI) Info() llm.ProviderInfo { return o.info }
    ```

    ```go
    package openai

    import (
        "errors"
        "net/http"
        "os"
        "time"

        "github.com/costa92/llm-agent/llm"
        openai "github.com/openai/openai-go/v3"
        "github.com/openai/openai-go/v3/option"
    )

    type config struct {
        apiKey       string
        model        string
        baseURL      string
        httpClient   *http.Client
        timeout      time.Duration
        organization string
    }

    // Option configures an OpenAI adapter at construction time. Per D-02 of
    // .planning/phases/01-walking-skeleton-generate/01-CONTEXT.md, every
    // adapter exposes the same five universal options plus a small set of
    // provider-specific extras (here: WithOrganization).
    type Option func(*config)

    // WithModel sets the bound model (REQUIRED — New returns error if empty).
    func WithModel(m string) Option { return func(c *config) { c.model = m } }

    // WithAPIKey overrides the OPENAI_API_KEY env-var fallback.
    func WithAPIKey(k string) Option { return func(c *config) { c.apiKey = k } }

    // WithBaseURL points the SDK at a custom endpoint (httptest, Azure, proxies).
    func WithBaseURL(u string) Option { return func(c *config) { c.baseURL = u } }

    // WithHTTPClient injects a custom transport (retries, OTel wrap, etc.).
    func WithHTTPClient(h *http.Client) Option { return func(c *config) { c.httpClient = h } }

    // WithTimeout sets a request-level timeout (separate from ctx).
    func WithTimeout(d time.Duration) Option { return func(c *config) { c.timeout = d } }

    // WithOrganization sets the OpenAI-Organization header (provider-specific).
    func WithOrganization(o string) Option { return func(c *config) { c.organization = o } }

    // New builds an OpenAI adapter. WithModel is required — returns error otherwise.
    // API key falls back to OPENAI_API_KEY env var when WithAPIKey is not given.
    func New(opts ...Option) (*OpenAI, error) {
        cfg := config{}
        for _, opt := range opts {
            opt(&cfg)
        }
        if cfg.model == "" {
            return nil, errors.New("openai: WithModel is required")
        }
        if cfg.apiKey == "" {
            cfg.apiKey = os.Getenv("OPENAI_API_KEY")
        }
        var sdkOpts []option.RequestOption
        if cfg.apiKey != "" {
            sdkOpts = append(sdkOpts, option.WithAPIKey(cfg.apiKey))
        }
        if cfg.baseURL != "" {
            sdkOpts = append(sdkOpts, option.WithBaseURL(cfg.baseURL))
        }
        if cfg.httpClient != nil {
            sdkOpts = append(sdkOpts, option.WithHTTPClient(cfg.httpClient))
        }
        if cfg.organization != "" {
            sdkOpts = append(sdkOpts, option.WithHeader("OpenAI-Organization", cfg.organization))
        }
        if cfg.timeout > 0 {
            sdkOpts = append(sdkOpts, option.WithRequestTimeout(cfg.timeout))
        }
        client := openai.NewClient(sdkOpts...)
        return &OpenAI{
            client: &client,
            info: llm.ProviderInfo{
                Provider: "openai",
                Model:    cfg.model,
                Capabilities: llm.Capabilities{
                    Tools:             false, // Pitfall A — Phase 1 forbids
                    Embeddings:        false,
                    StructuredOutputs: false,
                    PromptCaching:     false,
                },
            },
        }, nil
    }
    ```

    ```go
    package openai

    import (
        "github.com/costa92/llm-agent/llm"
        openai "github.com/openai/openai-go/v3"
    )

    // toSDKRequest maps llm.Request → openai.ChatCompletionNewParams.
    //
    // Phase 1 boundary (Pitfall A): Request.Tools is intentionally NOT mapped.
    // Tool calling lands in Phase 3 with full ToolCaller / per-tool-call indexing.
    func (o *OpenAI) toSDKRequest(req llm.Request) openai.ChatCompletionNewParams {
        msgs := make([]openai.ChatCompletionMessageParamUnion, 0, len(req.Messages)+1)
        if req.SystemPrompt != "" {
            msgs = append(msgs, openai.SystemMessage(req.SystemPrompt))
        }
        for _, m := range req.Messages {
            switch m.Role {
            case "user":
                msgs = append(msgs, openai.UserMessage(m.Content))
            case "assistant":
                msgs = append(msgs, openai.AssistantMessage(m.Content))
            case "system":
                msgs = append(msgs, openai.SystemMessage(m.Content))
            }
        }
        p := openai.ChatCompletionNewParams{
            Model:    o.info.Model,
            Messages: msgs,
        }
        if req.MaxOutputTokens > 0 {
            p.MaxCompletionTokens = openai.Int(int64(req.MaxOutputTokens))
        }
        if req.Temperature != nil {
            p.Temperature = openai.Float(float64(*req.Temperature))
        }
        // NOTE: req.Tools intentionally NOT mapped — Phase 1 ignores tools (Pitfall A).
        return p
    }

    // fromSDKResponse maps openai.ChatCompletion → llm.Response with
    // Usage.Source=UsageReported (OpenAI always returns token counts on success).
    func (o *OpenAI) fromSDKResponse(c *openai.ChatCompletion) llm.Response {
        var text string
        if len(c.Choices) > 0 {
            text = c.Choices[0].Message.Content
        }
        finish := llm.FinishReasonUnknown
        if len(c.Choices) > 0 {
            finish = mapFinishReason(string(c.Choices[0].FinishReason))
        }
        return llm.Response{
            Text:         text,
            FinishReason: finish,
            Provider:     "openai",
            Model:        c.Model,
            Usage: llm.Usage{
                InputTokens:  int(c.Usage.PromptTokens),
                OutputTokens: int(c.Usage.CompletionTokens),
                TotalTokens:  int(c.Usage.TotalTokens),
                Source:       llm.UsageReported,
            },
        }
    }

    // mapFinishReason normalizes OpenAI's stop_reason into llm.FinishReason*.
    func mapFinishReason(s string) llm.FinishReason {
        switch s {
        case "stop":
            return llm.FinishReasonStop
        case "length":
            return llm.FinishReasonLength
        case "content_filter":
            return llm.FinishReasonContentFilter
        case "tool_calls":
            return llm.FinishReasonToolCalls
        case "function_call":
            return llm.FinishReasonFunctionCall
        default:
            return llm.FinishReasonUnknown
        }
    }
    ```

    **Step E — create `openai/errors.go`** (verbatim from RESEARCH.md lines 343-399):

    ```go
    package openai

    import (
        "context"
        "errors"
        "net"

        "github.com/costa92/llm-agent/llm"
        openai "github.com/openai/openai-go/v3"
    )

    // wrapErr converts openai-go SDK errors into llm.AuthError / llm.RateLimitError /
    // llm.InvalidRequestError / llm.TransientError per the HTTP-status mapping
    // documented in PROVIDER_AUTHORING.md (D-03 of Phase 1). The SDK error is
    // preserved in the Wrapped field for errors.Unwrap chain traversal.
    func wrapErr(err error) error {
        if err == nil {
            return nil
        }
        // ctx.Canceled propagates as-is per D-03.
        if errors.Is(err, context.Canceled) {
            return err
        }
        if errors.Is(err, context.DeadlineExceeded) {
            return &llm.TransientError{Provider: "openai", Wrapped: err}
        }

        var apiErr *openai.Error
        if errors.As(err, &apiErr) {
            switch apiErr.StatusCode {
            case 401, 403:
                return &llm.AuthError{Provider: "openai", Wrapped: err}
            case 429:
                reason := ""
                if apiErr.Type == "insufficient_quota" || apiErr.Code == "insufficient_quota" {
                    reason = "quota_exhausted"
                }
                return &llm.RateLimitError{
                    Provider:   "openai",
                    RetryAfter: apiErr.Headers().Get("Retry-After"),
                    Reason:     reason,
                    Wrapped:    err,
                }
            case 500, 502, 503, 504:
                return &llm.TransientError{Provider: "openai", Wrapped: err}
            default:
                if apiErr.StatusCode >= 400 && apiErr.StatusCode < 500 {
                    return &llm.InvalidRequestError{Provider: "openai", Wrapped: err}
                }
            }
        }

        // Network-level error (DNS, TCP reset, closed conn): retry-eligible.
        var netErr net.Error
        if errors.As(err, &netErr) {
            return &llm.TransientError{Provider: "openai", Wrapped: err}
        }
        return err
    }
    ```
