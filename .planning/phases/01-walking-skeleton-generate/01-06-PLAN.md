    ```go
    //go:build ollama_live

    // Package contract — nightly testcontainers integration test.
    //
    // This file is build-tagged 'ollama_live' so PR CI never runs it (Pitfall D
    // mitigation: cold-pull is 3-5 min, blowing past PR step timeouts). The
    // nightly-ollama-live.yml GitHub workflow runs it via:
    //   go test -tags ollama_live -timeout 30m ./internal/contract/...
    package contract

    import (
        "context"
        "os"
        "testing"

        "github.com/costa92/llm-agent-providers/ollama"
        tcollama "github.com/testcontainers/testcontainers-go/modules/ollama"
    )

    // TestGenerate_Ollama_Live spins up a real Ollama testcontainer, pre-pulls
    // the pinned model, and runs the conformance suite's Ollama happy-path
    // fixture against it. If this goes red, Ollama upstream has changed
    // something (model behavior or wire format) — actionable signal, never a
    // blocker for PR CI.
    //
    // Pinned defaults (overridable via env in the workflow):
    //   OLLAMA_TC_IMAGE=ollama/ollama:0.5.7
    //   OLLAMA_TC_MODEL=llama3.1:8b-instruct-q4_K_M
    func TestGenerate_Ollama_Live(t *testing.T) {
        if testing.Short() {
            t.Skip("skipping ollama-live test in -short mode")
        }
        ctx := context.Background()
        image := getenv("OLLAMA_TC_IMAGE", "ollama/ollama:0.5.7")
        model := getenv("OLLAMA_TC_MODEL", "llama3.1:8b-instruct-q4_K_M")

        container, err := tcollama.Run(ctx, image)
        if err != nil {
            t.Fatalf("tcollama.Run(%s): %v", image, err)
        }
        t.Cleanup(func() { _ = container.Terminate(ctx) })

        // Pre-pull the model. testcontainers-go does not have a startup-time
        // WithModel option (Q5 RESOLVED); Exec is the documented path.
        if _, _, err := container.Exec(ctx, []string{"ollama", "pull", model}); err != nil {
            t.Fatalf("ollama pull %s: %v", model, err)
        }

        baseURL, err := container.ConnectionString(ctx)
        if err != nil {
            t.Fatalf("ConnectionString: %v", err)
        }

        adapter, err := ollama.New(ollama.WithModel(model), ollama.WithBaseURL(baseURL))
        if err != nil {
            t.Fatalf("ollama.New: %v", err)
        }

        // Run the conformance suite's happy-path fixture against the live
        // container. The fixture's response_text is "Hi from Ollama!" — the
        // real model will produce SOMETHING different, so AssertGenerate's
        // text-match assertion will fail unless the fixture's expect.response_text
        // is empty. We load the fixture purely for the structural assertions
        // (Provider, Usage.Source, FinishReason normalization).
        f := LoadFixture(t, "ollama", "generate_happy_llama3.1-8b")
        // Real model output is non-deterministic — null out the text assertion
        // so AssertGenerate only checks structural invariants.
        f.Expect.ResponseText = ""
        f.Expect.UsageInputTokens = 0  // also non-deterministic against real model
        f.Expect.UsageOutputTokens = 0
        AssertGenerate(t, adapter, f)
    }

    func getenv(k, def string) string {
        if v := os.Getenv(k); v != "" {
            return v
        }
        return def
    }
    ```

    Constraints:
    - The build tag MUST be `//go:build ollama_live` (Go 1.17+ syntax; older `// +build` is incompatible).
    - Tag goes on line 1 with a blank line below before the `package` declaration.
    - The test mutates `f.Expect.ResponseText`/`UsageInputTokens`/`UsageOutputTokens` to nil out non-deterministic assertions — real model output cannot match a fixed string. We keep the structural assertions (Provider, FinishReason normalization, Usage.Source).
    - Do NOT use `httptest.Server` here — this test runs against the REAL container's URL.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Tampering | ollama/ollama image floating tag | mitigate | Pinned to `ollama/ollama:0.5.7` (NOT `:latest`) per CONTEXT.md security threat model. SHA pinning is preferable but breaks model-volume cache key — accept tag-pinning as v0.3 baseline; revisit at v0.4 if a compromised tag becomes a real concern. |

CI runner is ephemeral; even a poisoned model can only affect the test result (red workflow), not the production codebase. |
| T-06-03 | Information Disclosure | Docker socket access in CI | accept | testcontainers-go requires Docker socket (mounted by default on `ubuntu-latest`). The runner is ephemeral; nothing persistent leaks. |
| T-06-04 | Denial of Service | nightly cron + cold pull = 45min budget | mitigate | timeout-minutes: 45 caps the budget; actions/cache reduces warm-run to <5min; cron at 03:00 UTC = off-peak for GH Actions queue. |
