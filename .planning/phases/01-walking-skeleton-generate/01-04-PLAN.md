```go
type statusCapturingTransport struct {
    inner http.RoundTripper
    last  *int32 // atomic int32; 0 = no response yet
}
func (t *statusCapturingTransport) RoundTrip(req *http.Request) (*http.Response, error) {
    resp, err := t.inner.RoundTrip(req)
    if resp != nil {
        atomic.StoreInt32(t.last, int32(resp.StatusCode))
    }
    return resp, err
}
```

wrapErr reads the captured status:
```go
func (o *Ollama) wrapErr(err error) error {
    if err == nil { return nil }
    if errors.Is(err, context.Canceled) { return err }
    if errors.Is(err, context.DeadlineExceeded) {
        return &llm.TransientError{Provider:"ollama", Wrapped: err}
    }
    status := int(atomic.LoadInt32(o.lastStatus))
    msg := err.Error()
    switch {
    case status == 401 || status == 403:
        return &llm.AuthError{Provider:"ollama", Wrapped: err}
    case status == 404 && (strings.Contains(msg, "not found") || strings.Contains(msg, "not pulled")):
        return &llm.InvalidRequestError{Provider:"ollama", Wrapped: err}
    case status >= 500 && status < 600:
        return &llm.TransientError{Provider:"ollama", Wrapped: err}
    case status >= 400 && status < 500:
        return &llm.InvalidRequestError{Provider:"ollama", Wrapped: err}
    }
    var netErr net.Error
    if errors.As(err, &netErr) {
        return &llm.TransientError{Provider:"ollama", Wrapped: err}
    }
    return err
}
```

    ```go
    // Package ollama implements the Phase-1 ChatModel adapter against
    // github.com/ollama/ollama/api (Ollama's /api/chat endpoint, sync mode).
    //
    // Ollama is keyless and runs locally by default (http://localhost:11434).
    // The SDK is callback-based; this adapter calls Chat with Stream=&false so
    // the callback is invoked exactly once with the final response.
    package ollama

    import (
        "context"
        "errors"

        "github.com/costa92/llm-agent/llm"
        api "github.com/ollama/ollama/api"
    )

    // Compile-time: ChatModel ONLY — Phase 1 ignores tools / embeddings / structured outputs.
    var _ llm.ChatModel = (*Ollama)(nil)

    // Ollama is the /api/chat sync adapter. Construct via New(opts ...Option).
    type Ollama struct {
        client     *api.Client
        info       llm.ProviderInfo
        lastStatus *int32 // atomic; populated by statusCapturingTransport (see options.go)
    }

    // Generate calls /api/chat with Stream=false and maps the single-callback
    // response back into llm.Response. SDK errors are wrapped via wrapErr —
    // status code is recovered from the last captured HTTP response.
    func (o *Ollama) Generate(ctx context.Context, req llm.Request) (llm.Response, error) {
        sdkReq := o.toSDKRequest(req)
        var captured api.ChatResponse
        err := o.client.Chat(ctx, sdkReq, func(resp api.ChatResponse) error {
            captured = resp
            return nil
        })
        if err != nil {
            return llm.Response{}, o.wrapErr(err)
        }
        return o.fromSDKResponse(captured), nil
    }

    // Stream is a Phase-1 stub. Streaming lands in Phase 2 (CONF-03; OLL-02).
    func (o *Ollama) Stream(_ context.Context, _ llm.Request) (llm.StreamReader, error) {
        return nil, errors.New("ollama: streaming not implemented in Phase 1; use Generate")
    }

    // Info returns the bound (provider, model, capabilities) tuple. Phase 1
    // adapters report Capabilities.Tools=false; Ollama's per-model tool variation
    // lands in Phase 3 (per-model strategy table; OLL-03).
    func (o *Ollama) Info() llm.ProviderInfo { return o.info }
    ```

    ```go
    package ollama

    import (
        "errors"
        "net/http"
        "net/url"
        "os"
        "sync/atomic"
        "time"

        "github.com/costa92/llm-agent/llm"
        api "github.com/ollama/ollama/api"
    )

    type config struct {
        model      string
        baseURL    string
        httpClient *http.Client
        timeout    time.Duration
    }

    // Option configures an Ollama adapter at construction time. Per D-02 of
    // .planning/phases/01-walking-skeleton-generate/01-CONTEXT.md, every
    // adapter exposes the universal options. Ollama drops WithAPIKey (keyless)
    // and exposes WithHost as a friendlier alias for WithBaseURL.
    type Option func(*config)

    // WithModel sets the bound model (REQUIRED).
    func WithModel(m string) Option { return func(c *config) { c.model = m } }

    // WithBaseURL points the SDK at a custom endpoint (httptest, remote host).
    // Default: http://localhost:11434 or $OLLAMA_HOST.
    func WithBaseURL(u string) Option { return func(c *config) { c.baseURL = u } }

    // WithHost is an alias for WithBaseURL — Ollama-flavored ergonomics.
    func WithHost(u string) Option { return WithBaseURL(u) }

    // WithHTTPClient injects a custom transport. Note: this adapter installs
    // a status-capturing RoundTripper around whatever client is provided; if
    // you replace the transport entirely, status-code-based error classification
    // will degrade to "all errors are TransientError unless a net.Error".
    func WithHTTPClient(h *http.Client) Option { return func(c *config) { c.httpClient = h } }

    // WithTimeout sets a request-level timeout (separate from ctx).
    func WithTimeout(d time.Duration) Option { return func(c *config) { c.timeout = d } }

    // statusCapturingTransport wraps an inner http.RoundTripper and stores the
    // last response's StatusCode in an atomic *int32 so wrapErr can classify
    // errors when the underlying SDK returns plain `error` (Q3 RESOLVED).
    type statusCapturingTransport struct {
        inner http.RoundTripper
        last  *int32
    }

    func (t *statusCapturingTransport) RoundTrip(req *http.Request) (*http.Response, error) {
        resp, err := t.inner.RoundTrip(req)
        if resp != nil {
            atomic.StoreInt32(t.last, int32(resp.StatusCode))
        }
        return resp, err
    }

    // New builds an Ollama adapter. WithModel is required.
    func New(opts ...Option) (*Ollama, error) {
        cfg := config{}
        for _, opt := range opts {
            opt(&cfg)
        }
        if cfg.model == "" {
            return nil, errors.New("ollama: WithModel is required")
        }
        if cfg.baseURL == "" {
            cfg.baseURL = os.Getenv("OLLAMA_HOST")
        }
        if cfg.baseURL == "" {
            cfg.baseURL = "http://localhost:11434"
        }
        u, err := url.Parse(cfg.baseURL)
        if err != nil {
            return nil, errors.New("ollama: invalid base URL: " + cfg.baseURL)
        }

        // Wrap the user's HTTP client (or http.DefaultTransport) with the
        // status-capturing transport. lastStatus is shared with the *Ollama struct.
        lastStatus := new(int32)
        var inner http.RoundTripper = http.DefaultTransport
        var httpClient *http.Client
        if cfg.httpClient != nil {
            httpClient = cfg.httpClient
            if httpClient.Transport != nil {
                inner = httpClient.Transport
            }
        } else {
            httpClient = &http.Client{}
        }
        if cfg.timeout > 0 {
            httpClient.Timeout = cfg.timeout
        }
        httpClient.Transport = &statusCapturingTransport{inner: inner, last: lastStatus}

        client := api.NewClient(u, httpClient)
        return &Ollama{
            client: client,
            info: llm.ProviderInfo{
                Provider: "ollama",
                Model:    cfg.model,
                Capabilities: llm.Capabilities{
                    Tools:             false, // Pitfall A
                    Embeddings:        false,
                    StructuredOutputs: false,
                    PromptCaching:     false,
                },
            },
            lastStatus: lastStatus,
        }, nil
    }
    ```

    ```go
    package ollama

    import (
        "github.com/costa92/llm-agent/llm"
        api "github.com/ollama/ollama/api"
    )

    // toSDKRequest maps llm.Request → *api.ChatRequest with Stream=&false (sync mode).
    //
    // Ollama accepts role:"system" in messages (unlike Anthropic), so SystemPrompt
    // can be prepended as a system-role message without lifting.
    //
    // Phase 1 boundary (Pitfall A): Request.Tools is intentionally NOT mapped.
    func (o *Ollama) toSDKRequest(req llm.Request) *api.ChatRequest {
        msgs := make([]api.Message, 0, len(req.Messages)+1)
        if req.SystemPrompt != "" {
            msgs = append(msgs, api.Message{Role: "system", Content: req.SystemPrompt})
        }
        for _, m := range req.Messages {
            msgs = append(msgs, api.Message{Role: m.Role, Content: m.Content})
        }
        streamOff := false
        return &api.ChatRequest{
            Model:    o.info.Model,
            Messages: msgs,
            Stream:   &streamOff, // pointer-to-false disables streaming
            // NOTE: req.Tools intentionally NOT mapped — Phase 1 ignores tools (Pitfall A).
        }
    }

    // fromSDKResponse maps api.ChatResponse → llm.Response.
    //
    // Ollama returns prompt_eval_count + eval_count which we surface as
    // InputTokens / OutputTokens with Source=UsageReported (the local model
    // is the source of truth for its own tokenization).
    func (o *Ollama) fromSDKResponse(r api.ChatResponse) llm.Response {
        return llm.Response{
            Text:         r.Message.Content,
            FinishReason: mapOllamaDoneReason(r.DoneReason),
            Provider:     "ollama",
            Model:        r.Model,
            Usage: llm.Usage{
                InputTokens:  r.PromptEvalCount,
                OutputTokens: r.EvalCount,
                TotalTokens:  r.PromptEvalCount + r.EvalCount,
                Source:       llm.UsageReported,
            },
        }
    }

    // mapOllamaDoneReason normalizes Ollama's done_reason into llm.FinishReason*.
    func mapOllamaDoneReason(s string) llm.FinishReason {
        switch s {
        case "stop":
            return llm.FinishReasonStop
        case "length":
            return llm.FinishReasonLength
        default:
            return llm.FinishReasonUnknown
        }
    }
    ```

    **Step E — `ollama/errors.go`:** wrapErr reads the captured status (method on *Ollama because it needs access to lastStatus):

    ```go
    package ollama

    import (
        "context"
        "errors"
        "net"
        "strings"
        "sync/atomic"

        "github.com/costa92/llm-agent/llm"
    )

    // wrapErr converts Ollama SDK errors per the HTTP-status mapping in
    // PROVIDER_AUTHORING.md (D-03 of Phase 1) plus the Ollama-specific override:
    //   - 404 with "not found" / "not pulled" → *llm.InvalidRequestError
    //     (operator must run `ollama pull <model>`)
    //
    // Status code is read from o.lastStatus, populated by statusCapturingTransport
    // installed in options.go (Q3 RESOLVED — Ollama SDK returns plain `error`).
    func (o *Ollama) wrapErr(err error) error {
        if err == nil {
            return nil
        }
        if errors.Is(err, context.Canceled) {
            return err
        }
        if errors.Is(err, context.DeadlineExceeded) {
            return &llm.TransientError{Provider: "ollama", Wrapped: err}
        }

        status := int(atomic.LoadInt32(o.lastStatus))
        msg := err.Error()
        switch {
        case status == 401 || status == 403:
            return &llm.AuthError{Provider: "ollama", Wrapped: err}
        case status == 404 && (strings.Contains(msg, "not found") || strings.Contains(msg, "not pulled") || strings.Contains(msg, "model")):
            return &llm.InvalidRequestError{Provider: "ollama", Wrapped: err}
        case status == 429:
            return &llm.RateLimitError{Provider: "ollama", Wrapped: err}
        case status >= 500 && status < 600:
            return &llm.TransientError{Provider: "ollama", Wrapped: err}
        case status >= 400 && status < 500:
            return &llm.InvalidRequestError{Provider: "ollama", Wrapped: err}
        }

        // No HTTP status captured → likely network failure (closed listener, DNS).
        var netErr net.Error
        if errors.As(err, &netErr) {
            return &llm.TransientError{Provider: "ollama", Wrapped: err}
        }
        // Fallback: any error reaching here without status or net.Error is treated
        // as transient (the daemon may be transitioning, the SDK may have surfaced
        // a connection-level issue without a net.Error wrap).
        return &llm.TransientError{Provider: "ollama", Wrapped: err}
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| OLLAMA_HOST (env or option) → SDK → outbound HTTP (typically localhost) | User-controlled URL — usually local but could point at a remote |
| Ollama daemon response → fromSDKResponse → llm.Response.Text | Untrusted local daemon (or remote) content |
| statusCapturingTransport → wrapErr → llm.*Error | Status code crosses goroutine boundary via atomic |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | OLLAMA_HOST URL injection | mitigate | url.Parse validates; if user sets OLLAMA_HOST="http://malicious.example/" they're choosing their endpoint, which is the expected escape hatch. Documentation in README.md flags the local-default. |
| T-04-02 | Information Disclosure | wrapErr Error() output | mitigate | No API key in Ollama (keyless). Wrapped err.Error() contains the upstream JSON `{"error":"..."}` body — Phase 1 boundary documents this; consumers should not log err.Error() at debug level if Ollama is configured to upstream a hosted host. |
| T-04-03 | Tampering | concurrent atomic int32 race | mitigate | Concurrent calls to Generate from multiple goroutines on the same *Ollama instance share a single lastStatus pointer — a race could cause wrapErr to read a stale status from a different in-flight request. Mitigated: each Generate's status is written atomically and read atomically immediately after err. The only race window is "two Generates fail at exactly the same wall-clock instant" — accepted as a Phase 1 limitation; the wrapping is still typed correctly (only the Reason / status detail may be wrong). Plan 01-05's conformance suite does NOT exercise concurrent calls per adapter. |
| T-04-04 | Spoofing | local Ollama daemon impersonation | accept | If an attacker can run a process on localhost binding port 11434, they can return arbitrary content. This is the same trust model as any local-IPC service — out of scope for adapter-level mitigation. |
| T-04-05 | Tampering | Pull of github.com/ollama/ollama (large module) | mitigate | go mod tidy in Task 1 + sister-repo CI gate. Pinned at v0.23.2. The full module is heavy (~50MB) but Go's import-graph pruning keeps only /api in the build. |
