    ```go
    package anthropic

    import (
        "github.com/costa92/llm-agent/llm"
        anthropic "github.com/anthropics/anthropic-sdk-go"
    )

    func (a *Anthropic) toSDKRequest(req llm.Request) anthropic.MessageNewParams {
        msgs := make([]anthropic.MessageParam, 0, len(req.Messages))
        sysPrompt := req.SystemPrompt
        for _, m := range req.Messages {
            switch m.Role {
            case "user":
                msgs = append(msgs, anthropic.NewUserMessage(anthropic.NewTextBlock(m.Content)))
            case "assistant":
                msgs = append(msgs, anthropic.NewAssistantMessage(anthropic.NewTextBlock(m.Content)))
            case "system":
                // Pitfall C: system goes to top-level System field, NOT messages.
                if sysPrompt == "" {
                    sysPrompt = m.Content
                } else {
                    sysPrompt = sysPrompt + "\n\n" + m.Content
                }
            }
        }
        p := anthropic.MessageNewParams{
            Model:     anthropic.Model(a.info.Model),
            MaxTokens: 1024, // Anthropic API requires MaxTokens; default if Request.MaxOutputTokens unset
            Messages:  msgs,
        }
        if sysPrompt != "" {
            p.System = []anthropic.TextBlockParam{{Text: sysPrompt}}
        }
        if req.MaxOutputTokens > 0 {
            p.MaxTokens = int64(req.MaxOutputTokens)
        }
        // NOTE: req.Tools intentionally NOT mapped — Phase 1 ignores tools (Pitfall A).
        // NOTE: Temperature handling deferred — Anthropic SDK uses anthropic.Float helper;
        //       verify shape at task time via go doc anthropic.MessageNewParams.Temperature.
        return p
    }

    func (a *Anthropic) fromSDKResponse(m *anthropic.Message) llm.Response {
        var text string
        for _, block := range m.Content {
            // Phase 1: extract only text blocks; tool_use blocks ignored (Pitfall A).
            if block.Type == "text" {
                text += block.Text
            }
        }
        return llm.Response{
            Text:         text,
            FinishReason: mapAnthropicStopReason(string(m.StopReason)),
            Provider:     "anthropic",
            Model:        string(m.Model),
            Usage: llm.Usage{
                InputTokens:  int(m.Usage.InputTokens),
                OutputTokens: int(m.Usage.OutputTokens),
                TotalTokens:  int(m.Usage.InputTokens + m.Usage.OutputTokens),
                Source:       llm.UsageReported,
            },
        }
    }

    func mapAnthropicStopReason(s string) llm.FinishReason {
        // Anthropic: "end_turn", "max_tokens", "stop_sequence", "tool_use"
        switch s {
        case "end_turn", "stop_sequence":
            return llm.FinishReasonStop
        case "max_tokens":
            return llm.FinishReasonLength
        case "tool_use":
            return llm.FinishReasonToolCalls
        default:
            return llm.FinishReasonUnknown
        }
    }
    ```

    **errors.go** — Anthropic-specific status mapping. PATH A (preferred — `anthropic.Error` is publicly exposed):

    ```go
    package anthropic

    import (
        "context"
        "errors"
        "net"

        "github.com/costa92/llm-agent/llm"
        anthropic "github.com/anthropics/anthropic-sdk-go"
    )

    // wrapErr converts anthropic-sdk-go errors per the HTTP-status mapping in
    // PROVIDER_AUTHORING.md (D-03 of Phase 1) plus the Anthropic-specific overrides:
    //   - 529 overloaded_error  → *llm.RateLimitError (NOT TransientError)
    //   - 400 invalid_request_error → *llm.InvalidRequestError
    func wrapErr(err error) error {
        if err == nil {
            return nil
        }
        if errors.Is(err, context.Canceled) {
            return err
        }
        if errors.Is(err, context.DeadlineExceeded) {
            return &llm.TransientError{Provider: "anthropic", Wrapped: err}
        }

        var apiErr *anthropic.Error
        if errors.As(err, &apiErr) {
            switch apiErr.StatusCode {
            case 401, 403:
                return &llm.AuthError{Provider: "anthropic", Wrapped: err}
            case 429:
                return &llm.RateLimitError{
                    Provider:   "anthropic",
                    RetryAfter: apiErr.Headers().Get("Retry-After"),
                    Wrapped:    err,
                }
            case 529:
                // Anthropic-specific: overloaded_error is rate-limit-like, NOT transient.
                return &llm.RateLimitError{
                    Provider: "anthropic",
                    Reason:   "overloaded",
                    Wrapped:  err,
                }
            case 500, 502, 503, 504:
                return &llm.TransientError{Provider: "anthropic", Wrapped: err}
            default:
                if apiErr.StatusCode >= 400 && apiErr.StatusCode < 500 {
                    return &llm.InvalidRequestError{Provider: "anthropic", Wrapped: err}
                }
            }
        }

        var netErr net.Error
        if errors.As(err, &netErr) {
            return &llm.TransientError{Provider: "anthropic", Wrapped: err}
        }
        return err
    }
    ```

| HTTP response content blocks → fromSDKResponse → llm.Response.Text | Untrusted upstream content |
| Anthropic Beta-headers (via WithBetaHeader) → outbound request | User-controlled header value |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Information Disclosure | wrapErr Error() with Wrapped SDK error | mitigate | Anthropic SDK errors include the upstream error message (no Authorization headers). RateLimitError formatting same as plan 01-02. Plan 01-05's TestErrorString_NoSecretLeak test re-asserts. |
| T-03-02 | Tampering | WithBetaHeader user input → outbound HTTP header | mitigate | Header value is user-controlled but writes to an HTTP header is benign; the SDK URL-encodes / sanitizes the header. The risk is user setting `WithBetaHeader("\\r\\nAuthorization: ...")` to inject — Go's http package rejects invalid header chars at write time. Accept. |
| T-03-03 | Spoofing | API key sourcing | accept | Same as plan 01-02 — env-var fallback only, no file or URL-borne sourcing. |
| T-03-04 | Information Disclosure | TestGenerate_Anthropic_SystemTopLevel — captured request body in test logs | mitigate | Captured body is local to test process; assertions use strings.Contains, never log the full body. | T-03-05 | Tampering | go.sum integrity | mitigate | go mod tidy in Task 1 + sister-repo CI gate verifies tidy is clean. anthropic-sdk-go v1.41.0 pinned at task time. |
